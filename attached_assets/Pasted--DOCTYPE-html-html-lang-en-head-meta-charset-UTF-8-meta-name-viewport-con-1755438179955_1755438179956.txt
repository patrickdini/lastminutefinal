    <script>
        // Global state
        let currentSelection = { nights: 2, extended: false };

        // Image carousel functionality
        function showImage(cardId, imageIndex) {
            const carousel = document.getElementById(`carousel${cardId}`);
            const dots = document.querySelectorAll('.dot');
            
            carousel.style.transform = `translateX(-${imageIndex * 50}%)`;
            
            dots.forEach((dot, index) => {
                dot.classList.toggle('active', index === imageIndex);
            });
        }

        // Timeline Extension
        function selectDate(cardId, dateId) {
            console.log(`Selected date: ${dateId} for card ${cardId}`);
            
            // Handle extension logic
            if (dateId === 'aug20') {
                // Toggle extension for Aug 20
                const circle = event.target;
                if (circle.classList.contains('extension')) {
                    if (!currentSelection.extended) {
                        circle.classList.remove('extension');
                        circle.classList.add('selected');
                        currentSelection.nights = 3;
                        currentSelection.extended = true;
                        
                        updatePricing(3, 27.9);
                        showExtensionConfirmation('Added night before (Aug 20) - Total: 3 nights');
                    }
                }
            }
            
            // Visual feedback
            const clickedElement = event.target;
            if (clickedElement.classList.contains('extension')) {
                clickedElement.style.transform = 'scale(1.1)';
                setTimeout(() => {
                    clickedElement.style.transform = 'scale(1)';
                }, 200);
            }
        }

        // Helper functions
        function updatePricing(nights, newPrice) {
            const priceElement = document.getElementById('price1');
            const bookButton = document.getElementById('bookBtn1');
            const savingsElement = document.getElementById('savings1');
            
            if (priceElement) {
                priceElement.textContent = `${newPrice}M`;
            }
            
            if (bookButton) {
                bookButton.textContent = `BOOK OFFER (${nights} NIGHT${nights > 1 ? 'S' : ''})`;
            }
            
            if (savingsElement && nights !== 2) {
                const originalPrice = nights === 3 ? 36.6 : 24.4;
                const savings = (originalPrice - newPrice).toFixed(1);
                const percentage = ((savings / originalPrice) * 100).toFixed(1);
                savingsElement.textContent = `üí∞ You Save ${savings}M (${percentage}%) with this Offer!`;
            }
        }

        function showExtensionConfirmation(message) {
            const popup = document.getElementById('confirmationPopup');
            const overlay = document.getElementById('popupOverlay');
            const messageElement = document.getElementById('confirmationMessage');
            
            if (popup && overlay && messageElement) {
                messageElement.textContent = message;
                popup.classList.remove('hidden');
                overlay.classList.remove('hidden');
                
                // Auto-hide after 3 seconds
                setTimeout(() => {
                    hideConfirmation();
                }, 3000);
            }
        }

        function hideConfirmation() {
            const popup = document.getElementById('confirmationPopup');
            const overlay = document.getElementById('popupOverlay');
            
            if (popup && overlay) {
                popup.classList.add('hidden');
                overlay.classList.add('hidden');
            }
        }

        // Add interactive effects
        document.addEventListener('DOMContentLoaded', function() {
            const interactiveElements = document.querySelectorAll('.date-circle, .book-button');
            
            interactiveElements.forEach(element => {
                element.addEventListener('mouseenter', function() {
                    if (!this.classList.contains('unavailable')) {
                        this.style.transform = 'scale(1.05)';
                    }
                });
                
                element.addEventListener('mouseleave', function() {
                    if (!this.classList.contains('unavailable')) {
                        this.style.transform = 'scale(1)';
                    }
                });
            });
        });
    </script>
