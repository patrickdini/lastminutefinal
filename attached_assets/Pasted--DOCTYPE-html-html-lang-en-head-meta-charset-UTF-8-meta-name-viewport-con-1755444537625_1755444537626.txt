    <script>
        document.addEventListener('DOMContentLoaded', () => {
            // --- Carousel Logic ---
            const slidesContainer = document.querySelector('.carousel-slides');
            const slides = document.querySelectorAll('.carousel-slide');
            let currentIndex = 0;
            const slideInterval = 2000; // 2 seconds

            function showNextSlide() {
                currentIndex = (currentIndex + 1) % slides.length; // Loop back to the start
                slidesContainer.style.transform = `translateX(-${currentIndex * 100}%)`;
            }

            // Start automatic sliding
            let autoSlide = setInterval(showNextSlide, slideInterval);

            // --- Modal (Lightbox) Logic ---
            const modal = document.getElementById('imageModal');
            const modalImg = document.getElementById('modalImage');
            const closeModal = document.querySelector('.close');
            const prevButton = document.querySelector('.prev');
            const nextButton = document.querySelector('.next');
            const images = document.querySelectorAll('.carousel-slide img');
            let currentModalIndex = 0;

            // Function to open the modal
            function openModal(index) {
                currentModalIndex = index;
                modal.style.display = 'flex'; // Use flex to center the content
                modalImg.src = images[currentModalIndex].src;
                clearInterval(autoSlide); // Stop carousel when modal is open
            }

            // Function to close the modal
            function closeTheModal() {
                modal.style.display = 'none';
                autoSlide = setInterval(showNextSlide, slideInterval); // Restart carousel
            }

            // Function to show the next image in the modal
            function showNextModalImage() {
                currentModalIndex = (currentModalIndex + 1) % images.length;
                modalImg.src = images[currentModalIndex].src;
            }

            // Function to show the previous image in the modal
            function showPrevModalImage() {
                currentModalIndex = (currentModalIndex - 1 + images.length) % images.length;
                modalImg.src = images[currentModalIndex].src;
            }

            // Add click event listeners to each image
            images.forEach((img, index) => {
                img.addEventListener('click', () => {
                    openModal(index);
                });
            });

            // Event listeners for modal controls
            closeModal.addEventListener('click', closeTheModal);
            nextButton.addEventListener('click', showNextModalImage);
            prevButton.addEventListener('click', showPrevModalImage);
            
            // Close modal by clicking outside the image
            modal.addEventListener('click', (e) => {
                if (e.target === modal) {
                    closeTheModal();
                }
            });
        });
    </script>
