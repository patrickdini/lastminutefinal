    <script>
        // JavaScript for Carousel Functionality
        document.addEventListener('DOMContentLoaded', () => {
            const slides = document.querySelectorAll('.carousel-slide');
            const prevBtn = document.getElementById('prev-btn');
            const nextBtn = document.getElementById('next-btn');
            const dotsContainer = document.getElementById('dot-indicators');
            
            let currentSlide = 0;
            const totalSlides = slides.length;

            // Create dot indicators for each slide
            for (let i = 0; i < totalSlides; i++) {
                const dot = document.createElement('button');
                dot.classList.add('w-3', 'h-3', 'rounded-full', 'transition');
                dot.dataset.slide = i;
                dotsContainer.appendChild(dot);
            }
            
            const dots = dotsContainer.querySelectorAll('button');

            // Function to show a specific slide
            function showSlide(slideIndex) {
                // Ensure index is within bounds
                if (slideIndex >= totalSlides) {
                    slideIndex = 0;
                } else if (slideIndex < 0) {
                    slideIndex = totalSlides - 1;
                }

                // Hide all slides and deactivate all dots
                slides.forEach(slide => {
                    slide.style.opacity = '0';
                });
                dots.forEach(dot => {
                    dot.classList.remove('bg-brand-gold');
                    dot.classList.add('bg-white', 'bg-opacity-50');
                });

                // Show the target slide and activate its dot
                slides[slideIndex].style.opacity = '1';
                dots[slideIndex].classList.add('bg-brand-gold');
                dots[slideIndex].classList.remove('bg-white', 'bg-opacity-50');
                
                currentSlide = slideIndex;
            }

            // Event Listeners for buttons
            nextBtn.addEventListener('click', () => {
                showSlide(currentSlide + 1);
            });

            prevBtn.addEventListener('click', () => {
                showSlide(currentSlide - 1);
            });

            // Event Listeners for dots
            dots.forEach(dot => {
                dot.addEventListener('click', (e) => {
                    const slideIndex = parseInt(e.target.dataset.slide);
                    showSlide(slideIndex);
                });
            });

            // Initialize the carousel
            showSlide(0);
        });
    </script>
