        // Form submission handling
        document.getElementById('bookingForm').addEventListener('submit', function(e) {
            e.preventDefault();
            
            // Collect form data
            const formData = {
                firstName: document.getElementById('firstName').value,
                lastName: document.getElementById('lastName').value,
                email: document.getElementById('email').value,
                phone: document.getElementById('phone').value,
                transfer: document.getElementById('transfer').value,
                transferAddress: document.getElementById('transferAddress').value,
                transferLocation: document.getElementById('transferLocation').value,
                privateBoat: document.getElementById('privateBoat').checked
            };
            
            // Here you would typically send this data to your server
            console.log('Booking data:', formData);
            
            // Show success message (you can customize this)
            alert('Thank you! Your booking request has been submitted. We\'ll be in touch within 24 hours to confirm your villa experience.');
        });

        // Transfer dropdown logic
        document.getElementById('transfer').addEventListener('change', function() {
            const transferDetails = document.getElementById('transferDetails');
            const transferAddress = document.getElementById('transferAddress');
            const transferLocation = document.getElementById('transferLocation');
            
            if (this.value === 'yes') {
                transferDetails.style.display = 'block';
                transferAddress.required = true;
                transferLocation.required = true;
            } else {
                transferDetails.style.display = 'none';
                transferAddress.required = false;
                transferLocation.required = false;
                transferAddress.value = '';
                transferLocation.value = '';
            }
        });

        // Add some nice interaction feedback
        const inputs = document.querySelectorAll('input, select');
        inputs.forEach(input => {
            input.addEventListener('focus', function() {
                if (this.type !== 'checkbox') {
                    this.parentElement.style.transform = 'translateY(-2px)';
                }
            });
            
            input.addEventListener('blur', function() {
                if (this.type !== 'checkbox') {
                    this.parentElement.style.transform = 'translateY(0)';
                }
            });
        });
    </script>
